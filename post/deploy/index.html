Deploy.sh

创造一个 shell script 以使得 hugo 实现自动化部署

#!/bin/bash
echo -e "\033[0;32mDeploying updates to GitHub...\033[0m"
cd felixmon.github.io && rm -rf * && cd ..
rm -rf public
hugo -D
cp -rf ./public/* ./felixmon.github.io
cd felixmon.github.io
git add --all
msg="rebuilding the site `date`"
if [ $# -eq 1 ]
  then msg="$1"
fi
git commit -m "$msg"
git push --force origin master
cd ..

#!/bin/bash

echo -e "\033[0;32mDeploying updates to GitHub...\033[0m"

# Build the project.
hugo # if using a theme, replace with `hugo -t <YOURTHEME>`

# Go To Public folder
cd public
# Add changes to git.
git add .

# Commit changes.
msg="rebuilding site `date`"
if [ $# -eq 1 ]
  then msg="$1"
fi
git commit -m "$msg"

# Push source and build repos.
git push origin master

# Come Back up to the Project Root
cd ..
